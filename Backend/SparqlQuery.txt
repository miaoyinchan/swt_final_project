Get wikidata id of country

SELECT * WHERE {
    ?con osmt:place "country";
         osmt:name ?name;
  FILTER (?name = "Aruba")
  ?con osmt:wikidata ?data.
  }
  
  
Get city of attraction

SELECT * WHERE {
    ?con osmt:tourism "museum";
         osmt:name ?name;
         osmt:addr:country ?country;
         osmt:addr:postcode ?pos;
         osmt:addr:city ?city
  FILTER (?country = "GB")
  ?con osmt:wikidata ?data.
  }
